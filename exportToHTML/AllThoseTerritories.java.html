14       public AllThoseTerritories(){ 
15           this.setSize(1250, 650); 
16           this.setResizable(false);

23       private void initWorldMap() { 
24    
25           Map<String, Territory> TerrMap = new TreeMap<>(); 
26           //Karte einlesen von .map Datei 
27    
28           Scanner uinput = new Scanner(System.in); 
29           String currentline = uinput.nextLine(); 
30    
31           while (!(currentline == null) && !currentline.equals("")) { 
32               String[] line = currentline.split(" "); 
33    
34               String TerrName = ""; 
35               int temp = line.length; 
36    
37               switch (line[0]) { 
38                   case "patch-of":                                                        //adds new Polygon to draw to the Territory specified 
39                       int xPoly[]; 
40                       int yPoly[]; 
41    
42                       for (int i = 1; i < temp; i++) { 
43                           if (!Character.isDigit(line[i].charAt(0))) { 
44                               TerrName += " " + line[i]; 
45                           } else { 
46                               temp = i; 
47                           } 
48                       } 
49    
50                       xPoly = new int[(line.length - temp + 1) / 2]; 
51                       yPoly = new int[(line.length - temp + 1) / 2]; 
52    
53                       for (int i = 0; i < line.length - temp; i++) { 
54                           if (i % 2 == 0) 
55                               xPoly[i / 2] = Integer.parseInt(line[i + temp]); 
56                           else 
57                               yPoly[i / 2] = Integer.parseInt(line[i + temp]); 
58                       } 
59    
60                       /* 
61                       System.out.println(TerrName);       //debugging stuff 
62    
63                       for (int i : xPoly) { 
64                           System.out.print(i + " "); 
65                       } 
66                       System.out.println(); 
67                       for (int i : yPoly) { 
68                           System.out.print(i + " "); 
69                       } 
70                       */ 
71    
72                       if (!TerrMap.containsKey(TerrName)) { 
73                           System.out.println("CREATING NEW STUFF"); 
74                           //create new Territory and add it to the World Map 
75                           Territory terr = new Territory(TerrName); 
76                           terr.addField(new Polygon(xPoly, yPoly, xPoly.length)); 
77                           TerrMap.put(TerrName, terr); 
78                       } else { 
79                           TerrMap.get(TerrName).addField(new Polygon(xPoly, yPoly, xPoly.length)); 
80                       } 
81    
82                       break; 
83                   case "capital-of": 
84                       for (int i = 1; i < temp; i++) { 
85                           if (!Character.isDigit(line[i].charAt(0))) { 
86                               TerrName += " " + line[i]; 
87                           } else { 
88                               temp = i; 
89                           } 
90                       } 
91                       Point p = new Point(Integer.parseInt(line[temp]), Integer.parseInt(line [temp+1])); 
92                       TerrMap.get(TerrName).setCapital(p); 
93                       break; 
94                   case "neighbors-of": 
95                       //TODO 
96                       break; 
97                   case "continent": 
98                       //TODO 
99                       break; 
100                  default: 
101                      break; 
102              } 
103              currentline = uinput.nextLine(); 
104          } 
105   
106          this.add(new Panel(TerrMap)); 
107      }
